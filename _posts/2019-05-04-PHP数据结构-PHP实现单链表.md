class Node
{
    public $id;
    public $data;
    public $next;

    public function __construct($id,$data)
    {
        $this->id   = $id;
        $this->data = $data;
        $this->next = null;
    }
}

class SingleLinkedList
{
    private $header = null;
    private $id = 0;

    public function __construct($data)
    {
        $this->header = new Node($this->id,$data);
    }

    public function add($data)
    {
        $this->id  = $this->id + 1;
        $newNode = new Node($this->id,$data);
        $current = $this->header;
        while ($current->next!=null){
            $current = $current->next;
        }
        $current->next = $newNode;

    }

    public function delete($id)
    {
        $Node = $this->findById($id);
        if($this->header==$Node){
            $this->clear();
        }else{
            $Prev = $this->findPrev($id);
            $Prev->next = $Node->next;
        }
    }

    public function update($id,$data)
    {
        $Node = $this->findById($id);
        $Node->data = $data;
    }

    public function findById($id)
    {
        $current = $this->header;
        while ($current->next!=null)
        {
            if($current->id == $id){
                return $current;
            }
            $current = $current->next;

        }
        exit('未找到该元素');

    }

    private function findPrev($id)
    {
        $Node = $this->findById($id);
        $current = $this->header;
        while ($current->next!=null)
        {
            if($current->next == $Node){
                return $current;
            }
            $current = $current->next;

        }
        exit('未找到该元素');

    }

    public function findByData($data)
    {
        $result = array();
        $current = $this->header;
        while ($current->next!=null)
        {
            if($current->data == $data){
                $result[] = $current;
            }
            $current = $current->next;

        }
        if(empty($result)) exit('未找到该元素');
        return $result;

    }

    public function clear(){
        $this->header = null;
        $this->id = 0;
    }

    public function P(){

        $result = array();
        $current = $this->header;
        while ($current->next!=null)
        {
            $result[] = array(
                'id' => $current->id,
                'dtata'=>$current->data
            );
            $current = $current->next;

        }
        $result[] = array(
            'id' => $current->id,
            'dtata'=>$current->data
        );

        print_r($result);
    }

}

$list = new SingleLinkedList('header');
$list->add('add');
$list->add('add1');
$list->add('add2');
$list->delete(1);
$list->update(2,'update');
$list->P();
